    指定字符串位置查找

     1 #指定字符查找
 2 s = 'F:/my_pycharm/pycharm_project/CSV表格/10.csv'
 3 
 4 print(s.find('/'))     # 2，  第一个/在2位置
 5 print(s.rfind('/'))    # 35， 最后一个/在35位置
 6 
 7 print(s.index('/'))    # 2，  第一个/在2位置
 8 print(s.index('/',3))  # 13， 从3位开始查找,第一个/在13位置
 9 
10 print(s.count('/',4))  # 3，  从4位开始计算,/出现了3次
11 
12 print(s[:s.rfind('/')]) # 输出为 F:/my_pycharm/pycharm_project/CSV表格
