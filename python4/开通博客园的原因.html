    #!usr/bin/env python
# -*- coding:utf-8 -*-


class MyBlogger:
    def __init__(self,name,web):
        self.name = name
        self.web = web
        print('我的博客名:%s' % self.name)
        print('我的博客网页:%s' % self.web)

    @staticmethod
    def cause(args):
        '''

        :param args:元组
        :return:
        '''
        print('开通博客园的原因:')
        if args:
            for index,i in enumerate(args, 1):
                print('%d、%s\n' %(index, i))


mb = MyBlogger('wengxiaobin95','https://www.cnblogs.com/wengxiaobin/')
cause = ('个人很喜欢Python的鸭子模型和封装','想分享一些关于学习的一些心得',
       '我记得学习面向对象的时候,印象最深刻的是一切皆对象,所以博客会尽量用类和对象的形式展示',
       '用类和对象,别人复制我的类,生成对象就可以使用了,挺方便的')
mb.cause(cause)
