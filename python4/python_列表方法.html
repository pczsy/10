    1.在列表后面追加元素

    user = []
while True:
    name = input("输入名字：")
    # 在列表后面追加元素
    user.append(name)
    print(user)

    2.在列表中插入元素

    user = ['张三','李四','王雾']
# 在"1"位置的前面插入“wdc”（在指定索引位置插入元素）
user.insert(1,'wdc')
print(user)

    3.在列表中删除元素

    user = ['张三','李四','王雾']
# 在列表中删除指定字段的元素
user.remove('张三')
print(user)

    user = ['张三','李四','王雾']
    # 在列表中删除指定索引位置的元素
    user.pop(1)
    print(user)

    user = ['张三','李四','王雾']
    # 默认删除最后一个元素
    user.pop()
    print(user)

    4.清空列表

    user = ['张三','李四','王雾']
# 清空列表
user.clear()
print(user)

     5.列表的追加

    aList = [123, 'xyz', 'zara', 'abc', 123]
bList = [2009, 'manni']
# 在aList列表的末尾一次追加bList中的值
aList.extend(bList)

print(aList)
