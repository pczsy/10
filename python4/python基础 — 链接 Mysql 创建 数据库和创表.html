    import pymysql

try:
    host='xxx'
    user='xxx'
    passwd='xxx'
    db='test01'
    port=3306
    Table_name='kaka5'

    #  链接到服务器
    db = pymysql.connect(host, user, passwd, db, port)
    # 创建游标，对数据进行操作
    cursor = db.cursor()
    # 删除已经存在的同名表格
    sql1 = 'DROP TABLES IF EXISTS ' + Table_name
    cursor.execute(sql1)
    # 使用SQL语句创建表
    sql2 ='create table ' +  Table_name + " (FIRSTNAME CHAR(20),LASTNAME CHAR(20),AGE INT,SEX CHAR(1),INCOME FLOAT)"
    # 执行上面的sql语句
    cursor.execute(sql2)
    # 创建多行数据 （已经测试过了，没问题）
    sql3="insert into " + Table_name + "(FIRSTNAME, LASTNAME, AGE, SEX, INCOME) VALUES ('J', 'dana', 18, 'M', '100000'), ('P', 'dana', 18, 'M', '100000'), ('huang', 'laoban', 19, 'M', '9999')"
    cursor.execute(sql3)
    # 提交执行
    db.commit()
    # 关闭链接服务器
    db.close()
    print("创建完成")

except Exception as e:
    print("创建失败" + str(e))
