class Animal(object):
    def run(self):
        print('Animal is running')
    #动物类

class Dog(Animal):
    def run(self):
        print('dog is Running...')
    def eat(self):
        print('Eating meat...')
    #狗类
class Cat(Animal):
    def run(self):
        print('Cat is running')
    #猫类
dog = Dog()
dog.run()
dog.eat()

cat = Cat()
cat.run()

run_twice(Dog())
