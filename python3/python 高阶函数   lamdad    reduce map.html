    # 求和
    def get_sum(l):
        rest = 0
        for i in l:
            rest += i
        return rest

    def get_sum_use_py(l):
        return sum(l)


    from functools import  reduce
    def f(m,n):
        return m + n

    def get_sum_use_reduce(l):
        return reduce(f,l)


    def get_sum_use_lamdad(l):
        return reduce(lambda m , n : m + n,l)


    if __name__ == '__main__':
        l = [1,2,4,6,7,8,9]
