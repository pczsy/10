    # 异常处理
'''异常处理格式
try:
    程序
except Exception as 异常名称:
    异常处理部分
'''
try:
    for i in range(0,10):
        print(i)
        if (i==4):
            print(kkk)
    print("hello")
except Exception as err:
    print(err)
