    #乘法表
# 从x开始，直到y，即区间[x, y)中的整数，注意：x < y，否则range中无元素
for i in range(1,10):
    for j in range(1,i+1):
        print(str(i)+"*"+str(j)+"="+str(i*j),end="  ")#end=""表示不换行输出
    print()

print("==========================================================================")
#逆向乘法表
# range(x, y, step)：从x开始，直到y，步长为step，即 x <= x + n * step < y，其中n为1开始的整数
for x in range(9,0,-1):
    for y in range(x,0,-1):
        print(str(x)+"*"+str(y)+"="+str(x*y),end="  ")#end=""表示不换行输出
    print()
